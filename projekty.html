Algorytm sortowania bogosort

Algorytm ten losuje kolejność elementów w tablicy, a następnie sprawdza, czy tablica jest posortowana, jeżeli tak, to zwraca tablicę, a w przeciwnym wypadku powtarza poprzednie czynności.

Pokaż 
import random
n=int(input("Length of a table: "))
i=0
table = []
while(i<n):
	table.append(random.randint(0,9))
	i+=1
print(table)
print("Sort: ")
def isInOrder(table):
	flag = 0
	temp = 1
	while temp < n:
		if (table[temp] < table[temp - 1]):
			flag = 1
		temp += 1
	if (not flag):
		return True
	return False

iteration=0
while True:
	if isInOrder(table):
		break
	else:
		iteration += 1
		random.shuffle(table)
		print(iteration)
		print(table)
